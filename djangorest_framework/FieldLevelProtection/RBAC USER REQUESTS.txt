UNDERSTANDING STRUCTURES:

a.) class AssignUserRole(ViewSet): 
    👉 This handles user actions
    (e.g. “I want to become a host”)

b.) class HostRequestViewSet(ModelViewSet):
    👉 This handles admin workflow
    (e.g. approve/reject requests)

❌ BIG MISTAKE TO AVOID: Don’t “insert” one inside the other.
    👉 That would mix:
        -user logic ❌
        -admin logic ❌


1.) FROM USER_SIDE:


from rest_framework.viewsets import ViewSet
from rest_framework.response import Response
from rest_framework.decorators import action
from rest_framework.permissions import IsAuthenticated
from core.models import HostRequest

class AssignUserRole(ViewSet):

    permission_classes = [IsAuthenticated]

    @action(detail=False, methods=['post'])
    def become_host(self, request):
        user = request.user

        if HostRequest.objects.filter(user=user, status='pending').exists():
            return Response(
                {"message": "You already have a pending request"},
                status=400
            )

        HostRequest.objects.create(user=user)

        return Response({'message': 'Verification Request Submitted'})

2.) ADDING NEW CLASS FOR ADMIN SIDE:

from rest_framework.viewsets import ModelViewSet
from rest_framework.permissions import IsAuthenticated
from useraccount.permissions import IsAdmin
from .models import HostRequest
from .serializers import HostRequestSerializer
from rest_framework.decorators import action
from rest_framework.response import Response

class HostRequestViewSet(ModelViewSet):
    queryset = HostRequest.objects.all()
    serializer_class = HostRequestSerializer

    def get_permissions(self):
        if self.action in ['approve', 'reject', 'list']:
            return [IsAuthenticated(), IsAdmin()]
        return [IsAuthenticated()]

    def perform_create(self, serializer):
        serializer.save(user=self.request.user)

    # 🔥 APPROVE
    @action(detail=True, methods=['post'])
    def approve(self, request, pk=None):
        host_request = self.get_object()

        if host_request.status != 'pending':
            return Response({"error": "Request already processed"}, status=400)

        host_request.status = 'approved'
        host_request.save()

        user = host_request.user
        user.role = 'host'
        user.is_host_approved = True
        user.save()

        return Response({"message": "Host approved successfully"})

    # 🔥 REJECT
    @action(detail=True, methods=['post'])
    def reject(self, request, pk=None):
        host_request = self.get_object()

        if host_request.status != 'pending':
            return Response({"error": "Request already processed"}, status=400)

        host_request.status = 'rejected'
        host_request.save()

        return Response({"message": "Host request rejected"})

3.)Adding Logic To Models:

class HostRequest(models.Model):
    STATUS_CHOICES = (
        ('pending', 'Pending'),
        ('approved', 'Approved'),
        ('rejected', 'Rejected'),
    )

    user = models.ForeignKey(User, on_delete=models.CASCADE)
    status = models.CharField(max_length=10, choices=STATUS_CHOICES, default='pending')

    def save(self, *args, **kwargs):
        # Check previous status
        if self.pk:
            old = HostRequest.objects.get(pk=self.pk)

            # If status changed to approved
            if old.status != 'approved' and self.status == 'approved':
                self.user.role = 'host'
                self.user.is_host_approved = True
                self.user.save()

        super().save(*args, **kwargs)

🧒 SIMPLE ANALOGY(Very Important):
    Before(1,2): Approval Logic Lives in One Room(API):
        -👉 If you don’t enter that room → nothing happens
    After:Approval logic lives in the building itself (model):
        -👉 No matter where you enter → it works
